  <script>
    function showMessage() {
      const nameInput = document.getElementById("herName");
      const name = nameInput.value.trim();
      const output = document.getElementById("output");
      const hint = document.getElementById("hint");

      if (!name) {
        output.innerText = "Please write your name first.";
        return;
      }

      const message = 
`Dear ${name},

I am really sorry for the misunderstanding.
I truly respect you and I never want to hurt you.

I can let myself be misunderstood,
I can take the blame on myself,
but I never want anything to hurt you or your name.

You are very important to me,
and I just hope you can feel how genuine this apology is.

– From me`;

      output.innerText = message;
      hint.innerText = "Thank you for reading this.";
      nameInput.disabled = true;
    }
  </script>
